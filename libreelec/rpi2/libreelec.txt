LibreELEC 11.0-nightly-20231008-3c43828 (Libre Embedded Linux Entertainment Center)

1. Squashfs

  1.1 Unsquashfs

	$ ver=11.0-nightly-20231008-3c43828
	$ wget http://releases.libreelec.tv/LibreELEC-RPi2.arm-$ver.img.gz
	$ mkdir ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver
	$ gzip -d LibreELEC-RPi2.arm-$ver.img.gz
	Mount LibreELEC image and copy SYSTEM to Downloads/libreelec/LibreELEC-RPi2.arm-$ver
	$ mv LibreELEC-RPi2.arm-$ver.img ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver/ && rm LibreELEC-RPi2.arm-$ver.img.gz
	$ cd ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver
	$ mv SYSTEM SYSTEM.orig
	$ unsquashfs SYSTEM.orig
	$ [ ! -d copy  ] && mkdir copy
	$ cd squashfs-root

  1.2 ALSA

		net.core.default_qdisc = pfifo_fast # fq_codel

  1.5 Mksquashfs

	$ cd ..
	$ [ -f SYSTEM ] && rm -f SYSTEM
	$ sudo mksquashfs squashfs-root SYSTEM -all-root -noappend

  1.6 Make image
	
	$ cp ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver/LibreELEC-RPi2.arm-$ver.img ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver/LibreELEC-RPi2.arm-$ver-audio.img
	$ fdisk -l ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver/LibreELEC-RPi2.arm-$ver-audio.img
	$ sudo mkdir /mnt/le
	$ sudo mount -o loop,offset=$(echo $((512*8192))) ~/Downloads/kodi/LibreELEC-RPi2.arm-$ver/LibreELEC-RPi2.arm-$ver-audio.img /mnt/le
	$ sudo rm /mnt/le/SYSTEM*
	$ sudo cp SYSTEM /mnt/le
	$ sync
	
	## SYSTEM to RAM
	$ sudo sed -i '1s/$/ toram/' /mnt/le/cmdline.txt
	$ cat /mnt/le/cmdline.txt
	
	$ sudo umount /mnt/le
	$ sudo rm -rf /mnt/le
